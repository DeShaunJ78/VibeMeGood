if (existing) {
  // Row exists — update timestamps

    .set({

      lastSyncedAt: new Date(),
      updatedAt: new Date(),
      ...(gameId ? { gameId } : {}),
    })
    .where(eq(ppLinesTable.id,
      existing.id));
} else {
  // New tier — insert fresh row
  const [newLine] = await db

      openedAt: new Date(),
      lastSyncedAt: new Date(),
