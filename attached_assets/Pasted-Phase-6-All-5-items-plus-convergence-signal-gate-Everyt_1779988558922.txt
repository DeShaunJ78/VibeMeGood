ITEM 1 — Aggressive Pre-Lock Scraper
In artifacts/api-server/src/lib/cron.ts

Declare at module level:
export let preLockActive = false;

Add a job that runs every minute:
cron.schedule("* * * * *", async () => {
  const now = new Date();
  const twoHoursOut = new Date(
    now.getTime() + 2 * 60 * 60 * 1000);
  const upcomingLocks = await db
    .select({ id: ppLinesTable.id })
    .from(ppLinesTable)
    .where(and(
      eq(ppLinesTable.isActive, true),
      gte(ppLinesTable.lockTime, now),
      lte(ppLinesTable.lockTime, twoHoursOut)
    ))
    .limit(1);
  if (upcomingLocks.length > 0) {
    preLockActive = true;
    await syncPpLines();
    broadcastSyncStatus("pp-lines-prelock", "success");
  } else {
    preLockActive = false;
  }
});

ITEM 5 — MLB Park Factors
In lib/projection/compute.ts after
computing mean add park factor for
MLB batting stats:

const PARK_FACTORS: Record<string, number> = {
  COL: 1.18, CIN: 1.12, BOS: 1.08,
  PHI: 1.06, TEX: 1.05, NYY: 1.03,
  TOR: 0.97, MIA: 0.95, OAK: 0.94,
  PIT: 0.93, NYM: 0.92, TB: 0.92,
  SF: 0.90, SD: 0.88, SEA: 0.88,
};

const battingStats = [
  "hits", "home runs", "total bases",
  "hits+runs+rbis", "rbis", "runs",
  "doubles", "triples", "stolen bases"
];

const isBatter = battingStats.some(s =>
  statType.toLowerCase().includes(s));

if (sport === "MLB" && isBatter
    && homeTeamAbbr) {
  mean = mean * (PARK_FACTORS[homeTeamAbbr]
    ?? 1.0);
}
